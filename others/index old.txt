    <script>
    function openPopup(eventTitle) {
         console.log('openPopup called with eventTitle:', eventTitle);
        const popup = document.getElementById('popup');
        const popupContent = document.getElementById('popup-content');

        fetch(eventTitle + '.html')  // Fetch the corresponding subevent page
            .then(response => response.text())
            .then(content => {
                 console.log('Fetched content:', content);
                popupContent.innerHTML = content;
                popup.style.display = 'flex';
            })
            .catch(error => {
                console.error('Error loading subevent content:', error);
            });
    }
    function closePopup() {
        const popup = document.getElementById('popup');
        popup.style.display = 'none';
    }
    document.querySelector('.popup-close').onclick = closePopup;
    function openLoginPopup() {
            const loginPopup = document.getElementById('login-popup');
            loginPopup.style.display = 'flex';
        }
        function closeLoginPopup() {
            const loginPopup = document.getElementById('login-popup');
            loginPopup.style.display = 'none';
        }
        // Automatically close login popup if user clicks outside of it
        window.addEventListener('click', function(event) {
        const loginPopup = document.getElementById('login-popup');
        if (event.target !== loginPopup && !loginPopup.contains(event.target)) {
            closeLoginPopup();
        }
    });
</script>
